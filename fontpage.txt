                searchSuggestions.innerHTML = '';
                suggestions.forEach(suggestion => {
                    const suggestionItem = document.createElement('div');
                    suggestionItem.textContent = suggestion;
                    searchSuggestions.appendChild(suggestionItem);
                });
            });
